  \begin{block}{Dataset \& Methodology}

    \begin{itemize}
      \item \textbf{18.9K} federal-level sites across \textbf{61 countries}; Internet Archive snapshots \textbf{1996--2025}; \textbf{7.3M} URLs extracted.
      \item \textbf{Tracker labeling:} EasyList + EasyPrivacy + \textbf{21 regional filter lists}; ownership via DuckDuckGo Tracker Radar (\textbf{365 of 1098} tracker domains mapped to \textbf{265 orgs}).
      \item \textbf{Statistics:} Mann--Kendall trend tests, Jaccard similarity, HHI, First-Observed Uptake (FOU).
    \end{itemize}
