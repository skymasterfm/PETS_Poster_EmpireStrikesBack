  \begin{block}{Implications \& Recommendations}

    \begin{itemize}
      \item \textbf{Default-off templates:} central web templates should disable third-party trackers by default with explicit opt-in.
      \item \textbf{Procurement safeguards:} contracts must mandate visibility into third-party components and data flows.
      \item \textbf{Privacy-preserving analytics:} prefer self-hosted or log-based analytics with periodic automated audits.
      \item \textbf{User-side:} block third-party cookies; use tracking protection or a dedicated browser profile for government tasks.
    \end{itemize}

  \end{block}
